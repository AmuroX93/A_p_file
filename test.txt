FONT_PATH = r"C:\Windows\Fonts\msyh.ttc"  # ä¸­æ–‡å­—ä½“è·¯å¾„

def generate_wordcloud(text):
    """æ ¹æ®æ–‡æœ¬ç”Ÿæˆè¯äº‘"""
    text_cut = " ".join(jieba.cut(text))
    wc = WordCloud(
        font_path=FONT_PATH,
        background_color='white',
        width=800,
        height=400
    ).generate(text_cut)
    
    wc.to_file("wordcloud.png")  # ä¿å­˜è¯äº‘å›¾
    plt.imshow(wc, interpolation='bilinear')
    plt.axis('off')
    plt.show()
    print("âœ… è¯äº‘å·²æ›´æ–°å¹¶ä¿å­˜ä¸º wordcloud.png\n")

def read_chat_file(path):
    """è¯»å–èŠå¤©è®°å½•æ–‡ä»¶"""

        with open(path, "r", encoding="utf-8") as f:
            return f.read()
    except Exception as e:
        print(f"âŒ è¯»å–æ–‡ä»¶å¤±è´¥: {e}")
        return ""
